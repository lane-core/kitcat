This module is meant to be used for typechecking of the Prim namespace,
the more compact export module is Prim.Prelude

```agda
{-# OPTIONS --safe --erased-cubical #-}

module Prim.Everything where

open import Prim.Type public
open import Prim.Data public
open import Prim.HCompU public
open import Prim.Kan public
open import Prim.Glue public
open import Prim.Sub public
```
